## Usage
```tsx

```
```tsx
<Badge>Badge</Badge>
```

## Next.js Link
Use the `badgeVariants` helper to make `<Link />` look like `<Badge />`.
```tsx
import Link from "next/link";
import { badgeVariants } from "~/components/ui/Badge";
```
```tsx
<Link href="/" className={badgeVariants({ variant: "outline" })}>
    Home
</Link>
```
