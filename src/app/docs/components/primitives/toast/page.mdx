## Installation
<Steps>

<StepTitle>Install the following dependencies:</StepTitle>
<StepContent>
```bash
pnpm add sonner next-themes
```
</StepContent>

<StepTitle>Copy and paste the following code into your project.</StepTitle>
<StepContent>
<ComponentSource id="toast" />
</StepContent>

<StepTitle>Add the Toaster component to your root layout.</StepTitle>
<StepContent>
```tsx
import "~/styles/globals.css";

import { Toaster } from "~/components/ui/Toast";

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  );
}
```
</StepContent>

</Steps>

## Usage
```tsx
import { toast } from "sonner";
```
```tsx
toast("Event has been created.")
```
