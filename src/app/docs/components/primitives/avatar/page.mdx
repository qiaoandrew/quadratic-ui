## Usage
```tsx
import { Avatar, AvatarFallback, AvatarImage } from "~/components/ui/Avatar";
```
```tsx
<Avatar>
  <AvatarImage src="https://github.com/qiaoandrew.png" alt="@qiaoandrew" />
  <AvatarFallback>AQ</AvatarFallback>
</Avatar>
```
